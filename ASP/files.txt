using Microsoft.AspNetCore.Mvc;
using SimpleResumeApp.Models;
using System.Collections.Generic;

namespace SimpleResumeApp.Controllers
{
    public class ResumeController : Controller
    {
        public IActionResult Index()
        {
            var myResume = new Resume
            {
                Name = "John Doe",
                Title = "Software Engineer",
                Summary = "A passionate developer with 3+ years of experience in web development.",
                Skills = new List<string> { "C#", "ASP.NET Core", "HTML", "CSS", "JavaScript" }
            };

            return View(myResume);
        }
    }
}
